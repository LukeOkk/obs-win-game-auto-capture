cmake_minimum_required(VERSION 3.21)

project(obs-win-game-auto-capture VERSION 0.1.0 LANGUAGES CXX C)

if(NOT WIN32)
    message(FATAL_ERROR "This plugin is Windows-only. Use the macOS build at the repo root for macOS.")
endif()

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# libobs paths come from build.ps1 (or set manually):
#   LIBOBS_INCLUDE_DIR -> .../obs-studio/libobs  (folder containing obs-module.h)
#   OBS_LIB            -> .../obs.lib            (import lib generated from obs.dll)
if(NOT LIBOBS_INCLUDE_DIR OR NOT EXISTS "${LIBOBS_INCLUDE_DIR}/obs-module.h")
    message(FATAL_ERROR
        "LIBOBS_INCLUDE_DIR not set or obs-module.h missing.\n"
        "Pass: -DLIBOBS_INCLUDE_DIR=/path/to/obs-studio/libobs")
endif()
if(NOT OBS_LIB OR NOT EXISTS "${OBS_LIB}")
    message(FATAL_ERROR
        "OBS_LIB not set or not found.\n"
        "Pass: -DOBS_LIB=/path/to/obs.lib (build.ps1 generates it from obs.dll)")
endif()

add_library(${PROJECT_NAME} MODULE)

target_sources(${PROJECT_NAME}
    PRIVATE
        src/plugin-main.cpp
        src/game-auto-source.cpp
        src/game-detect.cpp
        src/game-capture.cpp
)

# obsconfig.h is generated by a full obs-studio build; synthesize a minimal one
# so obs-module.h can be included without building OBS. Undefined @VARS@ become
# empty strings via configure_file(@ONLY) — harmless for a plugin.
set(OBS_RELEASE_CANDIDATE 0)
set(OBS_BETA 0)
configure_file(
    "${LIBOBS_INCLUDE_DIR}/obsconfig.h.in"
    "${CMAKE_BINARY_DIR}/generated/obsconfig.h"
    @ONLY
)

target_include_directories(${PROJECT_NAME} PRIVATE
    "${CMAKE_BINARY_DIR}/generated"
    "${LIBOBS_INCLUDE_DIR}"
)

target_link_libraries(${PROJECT_NAME}
    PRIVATE
        "${OBS_LIB}"
        windowsapp      # C++/WinRT runtime
        d3d11
        dxgi
        dxguid
        shell32         # SHQueryUserNotificationState
        mmdevapi        # ActivateAudioInterfaceAsync (process loopback)
        ole32
        runtimeobject
)

target_compile_definitions(${PROJECT_NAME} PRIVATE
    UNICODE
    _UNICODE
    NOMINMAX
    WIN32_LEAN_AND_MEAN
)

if(MSVC)
    target_compile_options(${PROJECT_NAME} PRIVATE /W3 /EHsc /utf-8)
endif()

# Produce "obs-win-game-auto-capture.dll" (no lib prefix).
set_target_properties(${PROJECT_NAME} PROPERTIES
    PREFIX ""
    OUTPUT_NAME "${PROJECT_NAME}"
    SUFFIX ".dll"
)

# Install layout matches OBS's per-user plugin folder; build.ps1 performs the
# actual copy (including the data/ locale files), so no install() rule here.
